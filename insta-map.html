#Grouping based on Git & Jupyter Book project 

# Predefined categories with file extensions and folder names
categories = {
    'Configuration Files': ['.yml', '.gitignore'],
    'Data Files': ['.csv'],
    'Documentation': ['.md'],
    'Images': ['.png', '.jpg', '.jpeg', '.gif'],
    'Notebooks': ['.ipynb'],
    'References and Requirements': ['.bib', '.txt'],
}

# Function to determine the category based on the file extension or folder name
def categorize_file(filename):
    for category, extensions in categories.items():
        if any(filename.endswith(ext) for ext in extensions):
            return category
    return 'Other'

def generate_grouped_tree(dir_path, prefix=""):
    category_groups = {category: [] for category in categories}  # Group files by categories
    category_groups['Other'] = []  # Catch-all for unclassified files and folders
    
    contents = sorted(os.listdir(dir_path))
    for item in contents:
        full_path = os.path.join(dir_path, item)
        if os.path.isfile(full_path):
            category = categorize_file(item)
            category_groups[category].append(item)
        elif os.path.isdir(full_path):
            if item not in exclude_dirs:  # Exclude certain directories
                category_groups['Other'].append(item)
    
    #
    for category, files in category_groups.items():
        if files:
            for file in files:
                print(f"{prefix}├── {file}")
                full_path = os.path.join(dir_path, file)
                if os.path.isdir(full_path):
                    new_prefix = prefix + '│   '
                    generate_grouped_tree(full_path, new_prefix)
